export function buildProfilesInfo(
  profiles: Profile[],
  lang: string,
): KeyboardProfileInfo[] {
  return profiles.map(({ network, url }) => {
    const icon = SOCIAL_ICONS[network] ?? "";
    const firstLetter = network[0].toUpperCase();
 
    return {
      id: network,
      section: "Social",
      title: `${lang == "en" ? "Visit" : "Visita mi"} ${network}`,
      url,
      icon,
      hotkey: `ctrl+${firstLetter}`,
    };
  });
}

export function dispatchCommandPaletteShortcut(
  document: Document,
  KeyboardEventCtor: typeof KeyboardEvent,
): void {
  const event = new KeyboardEventCtor("keydown", {
    key: "K",
    code: "KeyK",
    keyCode: 75,
    which: 75,
    ctrlKey: true,
    altKey: false,
    shiftKey: false,
    metaKey: false,
  });
  document.dispatchEvent(event);
}
